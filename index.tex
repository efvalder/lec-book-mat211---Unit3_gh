\section{Power Series definition}\label{power-series-definition}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Definition}, colbacktitle=quarto-callout-note-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-note-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

An infinite series of the form \[\sum_{n=0}^\infty a_n(x-x_0)^n\] where
\(x_0,a_0,a_1,a_2,\cdots\in\mathbb{R}\) is a \texttt{power\ series}

A power series \texttt{converges} \(\Leftrightarrow\)
\[\lim_{N\rightarrow \infty} \sum_{n=0}^N a_n(x-x_0)^n \quad\exists\]

\end{tcolorbox}

\subsection{Example 1. Converge this
limit?}\label{example-1.-converge-this-limit}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-tip-color}{\faLightbulb}\hspace{0.5em}{Example}, colbacktitle=quarto-callout-tip-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-tip-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\[\lim_{n\rightarrow \infty}\left( 1+x+\frac{x^2}{2}+\frac{x^3}{6}+\cdots+\frac{x^n}{n!}\right)=\]

\begin{quote}
\textbf{Solution}

\[\sum_{n=0}^\infty \frac{x^n}{n!}=1+x+\frac{x^2}{2}+\frac{x^3}{6}+\cdots=e^x\]
As \(n\) goes to infinite the limit do exist, then we say that the
infinite series do converge.

\begin{quote}
\textbf{Note}

\emph{Note:} Taylor Series\footnotemark{}

\(f(x)\) around \(a\) can be written as:

\[f(x)\approx \sum_0^\infty \frac{f^{(n)}(a)}{n!}(x-a)^n\]
\end{quote}
\end{quote}

\end{tcolorbox}

\footnotetext{Check Taylor series:
\url{https://en.wikipedia.org/wiki/Taylor_series}}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-tip-color}{\faLightbulb}\hspace{0.5em}{Example}, colbacktitle=quarto-callout-tip-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-tip-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\begin{quote}
\textbf{Solution:}

\section{Proof geometric infinite
sum}\label{proof-geometric-infinite-sum}

Easy peasy !

\[\text{if }|x|<1 \Rightarrow \boxed{\frac{1}{1-x}=\sum_{n=0}^\infty x^n}\]

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-caution-color}{\faFire}\hspace{0.5em}{Proof:}, colbacktitle=quarto-callout-caution-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-caution-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\[\sum_{n=0}^\infty x^n = \lim_{N\rightarrow\infty} \sum_{n=0}^N x^n= \lim_{N\rightarrow\infty} S_N\]

\[S_N=1+x+x^2+x^3+\cdots +x^N\]

Let's multiply by \(x\)

\[xS_N=x+x^2+x^3+\cdots +x^N+x^{N+1}\]

Now this is similar to \(S_N\), but is missing a one and have an extra
\(x^{N+1}\), then is we add one and subtract \(x^{N+1}\) we obtain
\(S_N\)

\[S_N=1+xS_N-x^{N+1}\]

Solving for \(S_N\):

\[S_N(1-x)=1-x^{N+1}\]

\[S_N=\frac{1}{1-x}-\frac{x^{N+1}}{1-x}\]

Now if \(|x|<1\) the last fraction will go to zero when
\(N\rightarrow \infty\).

\[\therefore \boxed{\sum_{n=0}^\infty x^n =\frac{1}{1-x};\quad |x|<1}\]

\end{tcolorbox}

\section{Theorem Power Series
convergence}\label{theorem-power-series-convergence}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Theorem}, colbacktitle=quarto-callout-note-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-note-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\begin{theorem}[Power Series
convergence]\protect\hypertarget{thm-Pow}{}\label{thm-Pow}

For any power series \[\sum_{n=0}^\infty a_n(x-x_0)^n\] exactly
\texttt{one} of these statements is true.

\begin{enumerate}
\def\labelenumi{(\roman{enumi})}
\tightlist
\item
  The power series converges only for \(x=x_0\)
\item
  The power series converges \(\forall x\in \mathbb{R}\)
\item
  \(\exists R>0\) such that the power series converges for \(|x-x_0|<R\)
  and diverges for \(|x-x_0|>R\)
\end{enumerate}

Such an \texttt{R} is called the \texttt{radius\ of\ convergence}

\end{theorem}

\end{tcolorbox}

\section{Radius of convergence}\label{radius-of-convergence}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Theorem}, colbacktitle=quarto-callout-note-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-note-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\begin{theorem}[Radius of
convergence]\protect\hypertarget{thm-radPow}{}\label{thm-radPow}

Suppose \(a_n\neq 0\)
\[\forall n\geq N\quad\text{ and}\quad \lim_{n\rightarrow\infty} \left|\frac{a_{n+1}}{a_n}\right|=L\]
Then the radius of convergence of \(\sum_{n=0}^\infty a_n(x-x_0)^n\) is
\[R=\frac{1}{L}\]

\end{theorem}

\end{tcolorbox}

\subsection{Example 3.}\label{example-3.}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-tip-color}{\faLightbulb}\hspace{0.5em}{Example}, colbacktitle=quarto-callout-tip-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-tip-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\[\sum_{n=0}^\infty \frac{x^n}{n!}\Rightarrow \boxed{x_0=0,\quad a_n=\frac{1}{n!}}\Rightarrow\]
Please observe that: \[(n+1)!=n!(n+1)\]

\[ \left|\frac{a_{n+1}}{a_n}\right|=\left|\frac{\frac{1}{(n+1)!}}{\frac{1}{n!}}\right|=\left|\frac{n!}{n!(n+1)}\right|=\left|\frac{1}{n+1}\right|\]

which goes to zero as \(n\rightarrow \infty\)

Then if L goes to zero, then R goes to infinity:

\[\boxed{\Rightarrow R=\frac{1}{L}=\infty}\]

\end{tcolorbox}

\subsection{Example 1}\label{example-1-1}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-tip-color}{\faLightbulb}\hspace{0.5em}{Example}, colbacktitle=quarto-callout-tip-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-tip-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

Find the power series solution for
\[L_y:=y''+y=0,\quad\text{in terms of}(x-x_0)\]

\begin{quote}
\textbf{Solution}

\[y=\sum_{n=0}^\infty a_n(x-x_0)^n\]
\[y'=\sum_{n=1}^\infty na_n(x-x_0)^{n-1}\]
\[y''=\sum_{n=2}^\infty n(n-1)a_n(x-x_0)^{n-2}\]

\[\Rightarrow\]
\[L_y:=\sum_{n=2}^\infty n(n-1)a_n(x-x_0)^{n-2}+\sum_{n=0}^\infty a_n(x-x_0)^n=0\]
Shifting index for \(y''\):
\[=\sum_{n=2-2}^{\infty-2} (n+2)(n+2-1)a_{n+2}(x-x_0)^{n+2-2}+\sum_{n=0}^\infty a_n(x-x_0)^n\]
\[=0\]

\[\sum_{n=0}^\infty \left[(n+2)(n+1)a_{n+2}+a_n\right](x-x_0)^n=0\]

\[\Leftrightarrow\]

\[(n+2)(n+1)a_{n+2}+a_n=0,\qquad \forall n\geq 0\]

\[\Leftrightarrow \boxed{a_{n+2}=-\frac{a_n}{(n+2)(n+1)}}\] (2\(^{nd}\)
order recurrence relation)\footnotemark{} \[a_0,a_1,a_2,a_3,\cdots\]
sequence depends upon both \(a_0,a_1\).

\(\boxed{\text{Case 1 (even index)}:}\)

\[ a_2=-\frac{a_0}{(0+2)(0+1)}=-\frac{a_0}{2\cdot 1}\]
\[a_4=-\frac{a_2}{(2+2)(2+1)}=-\frac{a_2}{4\cdot 3}=\frac{a_0}{4\cdot3\cdot 2\cdot 1}\]
\[a_6=-\frac{a_4}{(4+2)(4+1)}=-\frac{a_4}{6\cdot 5}=-\frac{a_0}{6!}\]
Now instead of a recurrence relation, we can state an explicit equation
for the even index:

\[\boxed{a_{2m}=(-1)^m\frac{a_0}{(2m)!},\qquad m=0,1,2,3,\cdots}\]\footnotemark{}

\(\boxed{\text{Case 2 (odd index)}:}\)

\[a_3=-\frac{a_1}{(1+2)(1+1)}=-\frac{a_1}{3\cdot 2}\]
\[a_5=-\frac{a_3}{(3+2)(3+1)}=\frac{a_1}{5\cdot 4\cdot 3\cdot 2}\]
\[a_7=-\frac{a_5}{(5+2)(5+1)}=-\frac{a_1}{7!}\]

\[\boxed{a_{2m+1}=(-1)^m\frac{a_1}{(2m+1)!},\qquad m=0,1,2,3,\cdots}\]

Final Solution:

\[y=\sum_{m=0}^\infty (-1)^m a_0 \frac{(x-x_0)^{2m}}{(2m)!}+\sum_{m=0}^\infty (-1)^m a_1 \frac{(x-x_0)^{2m+1}}{(2m+1)!}\]
\[y=a_0\sum_{m=0}^\infty (-1)^m  \frac{(x-x_0)^{2m}}{(2m)!}+a_1\sum_{m=0}^\infty (-1)^m  \frac{(x-x_0)^{2m+1}}{(2m+1)!}\]

Expanding cosine and sine function using taylor series:

\[\boxed{y=a_0\cos(x-x_0)+a_1\sin(x-x_0)}\]

Note: Check the Taylor series for cosine and sine
functions\footnotemark{}
\end{quote}

\end{tcolorbox}

\section{Theorem}\label{theorem-5}

\texttt{Laplace\ Transform}: \(\mathscr{L}\)

This chapter explores the Laplace transform technique, a fundamental
problem-solving approach in mathematics that simplifies complex problems
into manageable ones. By converting initial value problems with constant
coefficients into algebraic equations, the method provides a new
solution pathway. While sometimes offering an alternative to traditional
methods, Laplace transforms prove particularly efficient for physical
problems involving discontinuous forcing functions, outperforming
previous strategies.

\section{Linearity}\label{linearity}

\begin{itemize}
\tightlist
\item
  Recall the derivative operator
\end{itemize}

\begin{align}
\frac{d}{dx} f(x)&=f'(x)\\
\frac{d}{dx} (c_1f(x)+c_2g(x))&=c_1f'(x)+c_2g'(x)\\
&=c_1\frac{d}{dx}f(x)+c_2\frac{d}{dx}g(x)\\
\end{align} \(\frac{d}{dx}\) is a linear operator\footnote{un dia vi una
  vaca vestida de uniforme\ldots{}}.

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-caution-color}{\faFire}\hspace{0.5em}{More about Linear Operators:}, colbacktitle=quarto-callout-caution-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-caution-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

In the context of differential equations and Laplace transforms, a
linear operator is a mathematical tool that maps one function to another
in a way that is consistent with two key properties of linearity:
additivity and homogeneity. These properties allow for the
simplification and solution of differential equations using the powerful
techniques of linear algebra. When discussing Laplace transforms, linear
operators play a crucial role.

\begin{enumerate}
\def\labelenumi{\arabic{enumi}.}
\item
  \textbf{Additivity}: This property means that if you apply the linear
  operator to the sum of two functions, it's the same as applying the
  operator to each function individually and then adding the results.
  Mathematically, for a linear operator \(L\), and any two functions
  \(f\) and \(g\), this is written as \(L(f + g) = L(f) + L(g)\).
\item
  \textbf{Homogeneity} (or scalar multiplication): This property states
  that if you multiply a function by a scalar (a constant number) before
  applying the linear operator, it's the same as applying the operator
  to the function first and then multiplying the result by the scalar.
  Formally, for any function \(f\) and any scalar \(c\), this is
  expressed as \(L(c \cdot f) = c \cdot L(f)\).
\end{enumerate}

These properties are essential when working with differential equations
in the Laplace domain because the Laplace transform itself is a linear
operator. The Laplace transform converts differential equations, which
can be difficult to solve in their original form, into algebraic
equations, which are often much simpler to work with. This conversion
leverages the linearity of the Laplace transform to simplify the process
of solving complex differential equations, particularly those that
describe systems in engineering and physics.

For example, consider a linear differential equation with constant
coefficients. The application of the Laplace transform linearly maps
this differential equation into an algebraic equation in the Laplace
domain. The solution of this algebraic equation can then be found using
techniques from linear algebra. Finally, applying the inverse Laplace
transform (which is also a linear operator) to the solution in the
Laplace domain converts it back to the original domain, giving the
solution to the original differential equation.

In summary, linear operators, exemplified by the Laplace transform in
this context, are foundational to solving differential equations by
transforming them into a form where linear algebraic methods can be
applied, thereby greatly simplifying the process of finding solutions.

\section{Theorem: Laplace of the
derivative}\label{theorem-laplace-of-the-derivative}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Theorem}, colbacktitle=quarto-callout-note-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-note-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\begin{theorem}[Laplace Transform of the
derivative]\protect\hypertarget{thm-LTD}{}\label{thm-LTD}

If \(y(t)\) is \texttt{continuous} and of \texttt{exponential\ order}
for \(t\geq 0\) and if \(y'(t)\) is \texttt{piecewise\ continuous} for
\(t\geq 0\), then

\[\mathscr{L}\{y'(t)\}=sY(s)-y(0)\] where \(\mathscr{L}\{y(t)\}=Y(s)\)

\end{theorem}

\begin{quote}
\textbf{Proof:}

\[\mathscr{L}\{y'(t)\}=\int_0^\infty e^{-st} y'(t)\,dt\]

Integration by parts:\footnotemark{}

\[u=e^{-st}\rightarrow du=-se^{-st}\,dt\]
\[dv=y'(t)dt\rightarrow v=y(t)\]

\[\mathscr{L}\{y'(t)\}=\left[e^{-st}y(t)\right]\Big|_0^\infty-\int_0^\infty y(t)(-se^{-st})\,dt\]

\[\mathscr{L}\{y'(t)\}=\left[e^{-st}y(t)\right]\Big|_0^\infty+s\int_0^\infty y(t)e^{-st}\,dt\]

\[\mathscr{L}\{y'(t)\}=(0-y(0))+sY(s)\]
\[\mathscr{L}\{y'(t)\}=sY(s)-y(0)\]

\[\hspace{2in}\square\]
\end{quote}

\end{tcolorbox}

\texttt{of\ exponential\ order} for some \(C>0\), and if

\[\hspace{1.5in}=-\int_0^\infty e^{-st}(t\cdot f(t))\,dt\]
\[\hspace{2in}= -\mathscr{L}\{t\cdot f(t)\}\] \[\hspace{3in}\square\]
\end{quote}

\section{Theorem: Laplace of the product between displace unit function
and a displaced
function}\label{theorem-laplace-of-the-product-between-displace-unit-function-and-a-displaced-function}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Theorem}, colbacktitle=quarto-callout-note-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-note-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\begin{theorem}[]\protect\hypertarget{thm-Lpp}{}\label{thm-Lpp}

If \(F(s)=\mathscr{L}\{f(t)\}\) exists for \(s>C\) and if \(a>0\), then

\begin{equation}\phantomsection\label{eq-utf}{\boxed{\mathscr{L}\{u(t-a)f(t-a)\}=e^{-as}\mathscr{L}\{f(t)\}=e^{-as}F(s)},\qquad s>C}\end{equation}

\begin{quote}
\textbf{Proof:}

\[\mathscr{L}\{u(t-a)f(t-a)\}=\int_0^\infty e^{-st} u(t-a)f(t-a)\,dt\]

\[=\int_a^\infty e^{-st} f(t-a)\,dt\]

Change of variables: \[u=t-a\rightarrow u(a)=0\]

\[=\int_0^\infty e^{-s(u+a)} f(u)\,du\]

\[=e^{-sa}\int_0^\infty e^{-su} f(u)\,du\]

\[=e^{-as}F(s)\] \[=e^{-as}\mathscr{L}\{f(t)\}\] \[\hspace{2in}\square\]
\end{quote}

\end{theorem}

\end{tcolorbox}

\subsection{Example 3}\label{example-3-2}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-tip-color}{\faLightbulb}\hspace{0.5em}{Example}, colbacktitle=quarto-callout-tip-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-tip-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

Solve: \[
y'' + 4y = 
\begin{cases}
0, & 0 \le t < \pi \\
\sin t, & t \ge \pi
\end{cases}, \quad y(0) = 0, \quad y'(0) = 1
\]

\begin{quote}
\textbf{Solution:}

\textbf{Step 1:} Express the forcing function using the \textbf{unit
step function}

We rewrite the right-hand side as a single expression using the
\textbf{Heaviside (unit step) function} \(u(t - \pi)\):

\[
y'' + 4y = \sin t \cdot u(t - \pi)
\]

\textbf{Step 2:} Take Laplace Transforms

Let \(Y(s) = \mathcal{L}\{y(t)\}\). Then:

\begin{itemize}
\tightlist
\item
  \(\mathcal{L}\{y''(t)\} = s^2 Y(s) - s y(0) - y'(0) = s^2 Y(s) - 1\)
\item
  \(\mathcal{L}\{4y(t)\} = 4Y(s)\)
\end{itemize}

So the left-hand side becomes:

\[
s^2 Y(s) - 1 + 4Y(s)
\]

Now for the right-hand side:

\begin{itemize}
\tightlist
\item
  \[\mathcal{L}\{\sin t \cdot u(t - \pi)\} = \mathcal{L}\{-\sin(t-\pi) \cdot u(t - \pi)\}\]
\end{itemize}

Then

\[
\mathcal{L}\{\sin t \cdot u(t - \pi)\} = -e^{-\pi s} \cdot \mathcal{L}\{\sin t\} = -e^{-\pi s} \cdot \frac{1}{s^2 + 1}
\]

\textbf{Step 3:} Laplace Equation

Putting it all together:

\[
s^2 Y(s) - 1 + 4Y(s) = -\frac{e^{-\pi s}}{s^2 + 1}
\]

\[
(s^2 + 4)Y(s) = 1 - \frac{e^{-\pi s}}{s^2 + 1}
\]

\[
Y(s) = \frac{1}{s^2 + 4} - \frac{e^{-\pi s}}{(s^2 + 4)(s^2 + 1)}
\]

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\textbf{Step 4:} Inverse Laplace Using Partial Fractions

We already know:

\[
\mathcal{L}^{-1}\left\{\frac{1}{s^2 + 4}\right\} = \frac{1}{2} \sin(2t)
\]

Now consider:

\[
\frac{1}{(s^2 + 4)(s^2 + 1)} = \frac{As + B}{s^2 + 1} + \frac{Cs + D}{s^2 + 4}
\]

Multiply both sides by \((s^2 + 4)(s^2 + 1)\):

\[
1 = (As + B)(s^2 + 4) + (Cs + D)(s^2 + 1)
\]

Now expand both sides:

\begin{itemize}
\tightlist
\item
  \((As + B)(s^2 + 4) = As^3 + 4As + Bs^2 + 4B\)
\item
  \((Cs + D)(s^2 + 1) = Cs^3 + Cs + Ds^2 + D\)
\end{itemize}

Add everything:

\[
1 = (A + C)s^3 + (B + D)s^2 + (4A + C)s + (4B + D)
\]

Match coefficients:

\[
\begin{aligned}
s^3 &: A + C = 0 \\
s^2 &: B + D = 0 \\
s^1 &: 4A + C = 0 \\
s^0 &: 4B + D = 1
\end{aligned}
\]

From \(A + C = 0 \Rightarrow C = -A\)

Substitute into
\(4A + C = 0 \Rightarrow 4A - A = 0 \Rightarrow A = 0 \Rightarrow C = 0\)

Then from \(B + D = 0 \Rightarrow D = -B\)

And
\(4B + D = 1 \Rightarrow 4B - B = 1 \Rightarrow 3B = 1 \Rightarrow B = \frac{1}{3}, D = -\frac{1}{3}\)

So:

\[
\frac{1}{(s^2 + 4)(s^2 + 1)} = \frac{1/3}{s^2 + 1} - \frac{1/3}{s^2 + 4}
\]

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

Now, going back to \(Y(s)\):

\[
Y(s) = \frac{1}{s^2 + 4} - e^{-\pi s} \left( \frac{1}{3(s^2 + 1)} - \frac{1}{3(s^2 + 4)} \right)
\]

\begin{center}\rule{0.5\linewidth}{0.5pt}\end{center}

\textbf{Step 5:} Inverse Laplace Transform

\[
y(t) = \mathcal{L}^{-1}\left\{ \frac{1}{s^2 + 4} \right\} - \mathcal{L}^{-1}\left\{ e^{-\pi s} \left( \frac{1}{3(s^2 + 1)} - \frac{1}{3(s^2 + 4)} \right) \right\}
\]

Use the time-shifting property:

\begin{itemize}
\tightlist
\item
  \(\mathcal{L}^{-1} \left\{ \frac{1}{s^2 + 1} \right\} = \sin t\)
\item
  \(\mathcal{L}^{-1} \left\{ \frac{1}{s^2 + 4} \right\} = \frac{1}{2} \sin 2t\)
\end{itemize}

\[
\mathcal{L}^{-1} \left\{ e^{-\pi s} \cdot F(s) \right\} = u(t - \pi) \cdot f(t - \pi)
\]

So final solution:

\[
\boxed{
y(t) = \frac{1}{2} \sin 2t - u(t - \pi) \left[ \frac{1}{3} \sin(t - \pi) - \frac{1}{6} \sin 2(t - \pi) \right]
}
\]
\end{quote}

\section{Theorem: Laplace of a periodic
function}\label{theorem-laplace-of-a-periodic-function}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Theorem}, colbacktitle=quarto-callout-note-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-note-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

\begin{theorem}[]\protect\hypertarget{thm-pwc}{}\label{thm-pwc}

Let \(f(t)\) be piecewise continuous, periodic T, and of exponential
order for \(t\geq 0\), then

\begin{equation}\phantomsection\label{eq-pwc}{\boxed{\mathscr{L}\{f(t)\}=\frac{1}{1-e^{-sT}}\int_0^T e^{-st}f(t)\,dt}}\end{equation}

\begin{quote}
\textbf{Proof:}

\[\mathscr{L}\{f(t)\}=\int_0^\infty e^{-st} f(t)\,dt\]

\[=\int_0^T e^{-st} f(t)\,dt+\int_T^\infty e^{-st} f(t)\,dt\]

Change of variables: \[v=t-T\rightarrow v(T)=0,\quad dv=dt\]

\[=\int_0^T e^{-st} f(t)\,dt+\int_{v=0}^\infty e^{-s(v+T)} f(v+T)\,dv\]
and because the period is \(T\), then \(f(v+T)=f(v)\)

\[=\int_0^T e^{-st} f(t)\,dt+e^{-sT}\int_0^\infty e^{-sv} f(v)\,dv\]
\[\mathscr{L}\{f(t)\}=\int_0^T e^{-st} f(t)\,dt+e^{-sT}\mathscr{L}\{f(t)\}\]
\[\Rightarrow \mathscr{L}\{f(t)\}\left(1-e^{-sT}\right)=\int_0^T e^{-st} f(t)\,dt\]

\[\Rightarrow \mathscr{L}\{f(t)\}=\frac{1}{\left(1-e^{-sT}\right)}\int_0^T e^{-st} f(t)\,dt\]

\[\hspace{2in}\square\]
\end{quote}

\end{theorem}

\end{tcolorbox}

\chapter{L27: Convolution Integral
(8.6)}\label{l27-convolution-integral-8.6-1}

When applying the Laplace transform to solve differential equations, it
is common to encounter situations where the right-hand side of the
equation consists of the product of functions. For each of these
functions, we may already know the corresponding inverse Laplace
transform. This familiarity might lead us to incorrectly assume that the
inverse Laplace transform of the product of functions is simply the
product of their individual inverse transforms. However, this assumption
is incorrect. The Laplace transform of a product of functions does not
generally equal the product of their individual Laplace transforms. This
misconception underscores the importance of applying the correct
properties and methods when working with Laplace transforms in
differential equations.

\section{Inverse of the Product?}\label{inverse-of-the-product}

\[\boxed{\mathscr{L}^{-1}\{F(s)\,G(s)\}\neq \mathscr{L}^{-1}\{F(s)\}\mathscr{L}^{-1}\{G(s)\}}\]

\section{Convolution Theorem}\label{convolution-theorem}

\section{Example 1, solving IVP}\label{example-1-solving-ivp}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-tip-color}{\faLightbulb}\hspace{0.5em}{Example}, colbacktitle=quarto-callout-tip-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-tip-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

Suppose \(y=f(t)\) is the solution of the constant-coefficient
homogeneous initial value problem
\[y''+Py'+Qy=0,\qquad y(0)=0,\quad y'(0)=1\] Then the solution of the
inhomogeneous initial value problem

\[y''+Py'+Qy=g(t),\qquad y(0)=0,\quad y'(0)=0\]

is \[y(t)=(f*g)(t)\]

\begin{quote}
\textbf{Solution}

Taking the Laplace Transform of homogeneous eqn:

\[s^2Y-1+PsY+QY=0\] \[Y=\frac{1}{s^2+Ps+Q}\]
\[y=f(t)=\mathscr{L}^{-1}\left\{\frac{1}{s^2+Ps+Q}\right\}\] Taking
Laplace transform of non-homogeneous case

\[s^2Y+PsY+QY=G\] \[Y=\frac{G}{s^2+Ps+Q}\]
\[y(t)=\mathscr{L}^{-1}\left\{\frac{G}{s^2+Ps+Q}\right\}\]
\[=\mathscr{L}^{-1}\left\{G(s)F(s)\right\}\] \[=(f*g)(t)\]
\[\hspace{3in}\square\]
\end{quote}

\end{tcolorbox}

\begin{tcolorbox}[enhanced jigsaw, title=\textcolor{quarto-callout-note-color}{\faInfo}\hspace{0.5em}{Definition}, colbacktitle=quarto-callout-note-color!10!white, opacitybacktitle=0.6, opacityback=0, rightrule=.15mm, breakable, titlerule=0mm, colback=white, bottomtitle=1mm, colframe=quarto-callout-note-color-frame, toprule=.15mm, leftrule=.75mm, toptitle=1mm, arc=.35mm, bottomrule=.15mm, left=2mm, coltitle=black]

An equation of the form

is a Volterra integral equation. Here \(f\) and \(k\) are given
functions and \(y\) is unknown. Since the integral on the right is a
convolution integral, the convolution theorem provides a convenient
formula for solving it. Laplace!

\[Y = F + YK\] \[Y=\frac{F}{1-K}\]

\end{tcolorbox}

\section{}\label{section-2}

\begin{theorem}[]\protect\hypertarget{thm-deltaa}{}\label{thm-deltaa}

\[\boxed{\mathscr{L}\{\delta(t-a)\}=e^{-as}},\qquad a>0,\quad s>0\]

\end{theorem}

\textbf{Proof:}

\[\mathscr{L}\{\delta(t-a)\}=\int_0^\infty e^{-st}\delta(t-a)\,dt\]
\[=\lim_{\epsilon \rightarrow 0} \int_0^\infty \delta_\epsilon (t-a) e^{-st}\,dt\]

Change of variables: \(u=t-a\rightarrow du=dt\)

\[=\lim_{\epsilon \rightarrow 0} \int_{-a}^\infty \delta_\epsilon (u) e^{-s(u+a)}\,du\]

\[=e^{-sa}\lim_{\epsilon \rightarrow 0} \int_{-a}^\infty \frac{1}{2\epsilon} e^{-su}\,du\]

\(a>\epsilon\), and by the definition of \(\delta_\epsilon\):

\[=e^{-sa}\lim_{\epsilon \rightarrow 0} \int_{-\epsilon}^\epsilon \frac{1}{2\epsilon} e^{-su}\,du\]

\[=e^{-sa}\lim_{\epsilon \rightarrow 0} \left[\left(\frac{e^{-su}}{-2s}\right)\Big|_{-\epsilon}^\epsilon\right]\]
\[=e^{-sa}\lim_{\epsilon \rightarrow 0} \left(\frac{1}{-2s}\right)\left[e^{-s\epsilon} - e^{s\epsilon}\right]\]

\[=e^{-sa}\lim_{\epsilon \rightarrow 0} \left(\left[\frac{e^{s\epsilon} - e^{-s\epsilon}}{2s}\right]\right)\]\footnote{\url{https://en.wikipedia.org/wiki/Hyperbolic_functions}}

\[= e^{-sa}\lim_{\epsilon \rightarrow 0} \left(\frac{\sinh s\epsilon}{s\epsilon}\right)\]

\[= e^{-sa}\lim_{\epsilon \rightarrow 0} (\cosh s\epsilon)\]
\[=e^{-sa}\]

\[\qquad\square\]
